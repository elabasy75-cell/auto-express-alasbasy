  <script>
    let inventory = JSON.parse(localStorage.getItem('inventory')) || [];
    let invoiceItems = [];

    function login() {
      document.getElementById('loginPage').style.display = 'none';
      document.getElementById('mainApp').style.display = 'block';
      localStorage.setItem('isLoggedIn', 'true');
      loadInventory();
      updateItemSelect();
    }

    if (localStorage.getItem('isLoggedIn') === 'true') {
      document.getElementById('loginPage').style.display = 'none';
      document.getElementById('mainApp').style.display = 'block';
      loadInventory();
      updateItemSelect();
    }

    function showPage(n) {
      document.querySelectorAll('.card').forEach(c => c.classList.remove('active'));
      document.getElementById('page' + n).classList.add('active');
    }

    function loadInventory() {
      const table = document.getElementById('inventoryTable');
      table.innerHTML = `<tr><th>الصنف</th><th>الماركة</th><th>السعر</th><th>الكمية</th><th>حالة</th></tr>`;
      
      inventory.forEach((item, i) => {
        const isLow = item.qty < 5;
        table.innerHTML += `
          <tr class="${isLow ? 'low-stock' : ''}">
            <td>${item.name}</td>
            <td>${item.brand}</td>
            <td>${item.price}</td>
            <td>${item.qty}</td>
            <td>${isLow ? '⚠️ منخفض' : '✅ جيد'}</td>
          </tr>`;
      });
      updateItemSelect();
    }

    function addItem() {
      const name = document.getElementById('itemName').value;
      const brand = document.getElementById('itemBrand').value;
      const price = Number(document.getElementById('itemPrice').value);
      const qty = Number(document.getElementById('itemQty').value);

      if (name && brand && price && qty) {
        inventory.push({name, brand, price, qty});
        localStorage.setItem('inventory', JSON.stringify(inventory));
        loadInventory();
        alert('✅ تم إضافة القطعة');
      }
    }

    function updateItemSelect() {
      const select = document.getElementById('itemSelect');
      select.innerHTML = '<option value="">اختر قطعة من المخزون</option>';
      inventory.forEach((item, i) => {
        select.innerHTML += `<option value="${i}">${item.name} - ${item.brand} (${item.qty})</option>`;
      });
    }

    function addToInvoice() {
      const index = document.getElementById('itemSelect').value;
      const qty = Number(document.getElementById('invoiceQty').value);

      if (index === "" || qty <= 0) return alert("اختر قطعة وكمية");

      const item = inventory[index];
      if (item.qty < qty) return alert("الكمية غير كافية في المخزون!");

      invoiceItems.push({
        name: item.name,
        brand: item.brand,
        price: item.price,
        qty: qty
      });

      // خصم من المخزون
      item.qty -= qty;
      localStorage.setItem('inventory', JSON.stringify(inventory));
      
      renderInvoice();
      loadInventory();
    }

    function renderInvoice() {
      let html = `<tr><th>الصنف</th><th>السعر</th><th>الكمية</th><th>الإجمالي</th></tr>`;
      let total = 0;
      invoiceItems.forEach(item => {
        const subtotal = item.price * item.qty;
        total += subtotal;
        html += `<tr><td>${item.name}</td><td>${item.price}</td><td>${item.qty}</td><td>${subtotal}</td></tr>`;
      });
      document.getElementById('invoiceTable').innerHTML = html;
      document.getElementById('total').textContent = total;
    }

    function printInvoice() {
      if (invoiceItems.length === 0) return alert("الفاتورة فارغة");

      const customer = document.getElementById('customerName').value || "عميل عام";
      let printContent = `
        <div style="text-align:center; font-family:Arial; padding:20px;">
          <h1>أوتو إكسبريس العباسي</h1>
          <p>01027058242</p>
          <hr>
          <h2>فاتورة</h2>
          <p>العميل: ${customer}</p>
          <table style="width:100%; border-collapse:collapse;">
            <tr><th>الصنف</th><th>السعر</th><th>الكمية</th><th>الإجمالي</th></tr>
      `;

      let total = 0;
      invoiceItems.forEach(item => {
        const sub = item.price * item.qty;
        total += sub;
        printContent += `<tr><td>${item.name}</td><td>${item.price}</td><td>${item.qty}</td><td>${sub}</td></tr>`;
      });

      printContent += `</table><h3>الإجمالي: ${total} جنيه</h3><p>شكراً لتعاملك معنا</p></div>`;

      const printWin = window.open('', '_blank');
      printWin.document.write(printContent);
      printWin.document.close();
      printWin.print();
    }

    function searchInventory() {
      // يمكن تحسينه لاحقاً
      loadInventory();
    }
